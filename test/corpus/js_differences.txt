; produces an ERROR node. The `= arr` portion is recovered as a separate
; expression statement.

================================================================================

================================================================================

--------------------------------------------------------------------------------

; Ucode has no destructuring. A bare `[a, b] = [...]` is parsed as a subscript
; expression where `[a, b]` is the object and `= [...]` is invalid inside the
; subscript, so the parser produces an ERROR node around the `=` and its operand.

(program
  (expression_statement
    (subscript_expression
      object: (array
        (identifier)
        (identifier))
      (ERROR)
      index: (sequence_expression
        (number)
        (number)))))
