================================================================================
Raw text only
================================================================================

Hello, world!

--------------------------------------------------------------------------------

(markup
  (raw_text))

================================================================================
Braces that are not tag openers become raw text
================================================================================

{ not a tag } and {another}

--------------------------------------------------------------------------------

(markup
  (raw_text))

================================================================================
Expression tag
================================================================================

Hello, {{ name }}!

--------------------------------------------------------------------------------

(markup
  (raw_text)
  (expression_tag
    open: (expression_tag_open)
    (identifier)
    close: (expression_tag_close))
  (raw_text))

================================================================================
if alt-syntax
================================================================================
{% if (x): %}body{% endif %}
--------------------------------------------------------------------------------

(markup
  (if_alt_statement
    open: (statement_tag_open)
    condition: (parenthesized_expression
      (identifier))
    close: (statement_tag_close)
    (raw_text)
    end_open: (statement_tag_open)
    end_close: (statement_tag_close)))

================================================================================
for-in alt-syntax
================================================================================
{% for (let i in items): %}{{ items[i] }}{% endfor %}
--------------------------------------------------------------------------------

(markup
  (for_in_alt_statement
    open: (statement_tag_open)
    left: (identifier)
    right: (identifier)
    close: (statement_tag_close)
    body: (expression_tag
      open: (expression_tag_open)
      (subscript_expression
        object: (identifier)
        index: (identifier))
      close: (expression_tag_close))
    end_open: (statement_tag_open)
    end_close: (statement_tag_close)))

counting for alt-syntax

(markup

      right: (number))
    condition: (binary_expression

      right: (number))
    increment: (update_expression
      argument: (identifier))
    close: (statement_tag_close)
    body: (raw_text)
    end_open: (statement_tag_open)
    end_close: (statement_tag_close)))
